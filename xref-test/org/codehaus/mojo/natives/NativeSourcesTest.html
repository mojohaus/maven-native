156     public void testExcludes()
157         throws Exception
158     {
159 
160         new File( this.workDirectory, "someFile.c" ).createNewFile();
161         new File( this.workDirectory, "someFile.cpp" ).createNewFile();
162 
163         NativeSources source = new NativeSources();
164 
165         source.setDirectory( this.workDirectory );
166 
167         String[] includes = { "*.*" };
168         source.setIncludes( includes );
169         assertEquals( 2, source.getFiles().size() );
170 
171         String[] excludes = { "*.cpp" };
172         source.setExcludes( excludes );
173         assertEquals( 1, source.getFiles().size() );
174 
175         String[] excludes2 = { "*.c" };
176         source.setExcludes( excludes2 );
177         assertEquals( 1, source.getFiles().size() );
178 
179         String[] excludes3 = { "someFile.c" };
180         source.setExcludes( excludes3 );
181         assertEquals( 1, source.getFiles().size() );
182 
183         String[] excludes4 = { "someFile.c", "someFile.cpp" };
184         source.setExcludes( excludes4 );
185         assertEquals( 0, source.getFiles().size() );
