1   package org.codehaus.mojo.natives.javah;
2   
3   import java.io.File;
4   
5   import org.codehaus.plexus.PlexusTestCase;
6   import org.codehaus.plexus.util.cli.Commandline;
7   import static org.junit.Assert.*;
8   
9   public class JavahExecutableTest
10      extends PlexusTestCase
11  {
12      private JavahConfiguration config;
13  
14      public void setUp()
15          throws Exception
16      {
17          super.setUp();
18  
19          this.config = new JavahConfiguration();
20  
21          String[] classPaths = { "path1", "path2" };
22          config.setClassPaths( classPaths );
23  
24          String[] classNames = { "className1", "className2" };
25          config.setClassNames( classNames );
26  
27          config.setOutputDirectory( new File( getBasedir(), "target/native" ) );
28  
29      }
30  
31      public void testDefaultJavahExecutable()
32          throws Exception
33      {
34          JavahExecutable javah = new JavahExecutable();
35          Commandline cl = javah.createJavahCommand( config );
36  
37          File outputDir = new File( getBasedir(), "target/native" );
38  
39          assertEquals( "javah", cl.getLiteralExecutable() );
40          assertArrayEquals( new String[] { "-d", outputDir.getAbsolutePath(), "-classpath",
41                  "path1" + File.pathSeparator + "path2", "className1", "className2" }, cl.getArguments() );
42      }
43  
44      public void testConfiguredJavahExecutable()
45          throws Exception
46      {
47          File javaBin = new File( "/java/home/bin" );
48  
49          JavahExecutable javah = new JavahExecutable();
50          config.setJavahPath( javaBin );
51          Commandline cl = javah.createJavahCommand( config );
52  
53          File outputDir = new File( getBasedir(), "target/native" );
54  
55          assertEquals( javaBin.getAbsolutePath(), cl.getLiteralExecutable() );
56          assertArrayEquals( new String[] { "-d", outputDir.getAbsolutePath(), "-classpath",
57                  "path1" + File.pathSeparator + "path2", "className1", "className2" }, cl.getArguments() );
58      }
59  
60      public void testJavahExecutableDashoOption()
61          throws Exception
62      {
63          config.setFileName( "fileName" );
64          JavahExecutable javah = new JavahExecutable();
65          Commandline cl = javah.createJavahCommand( config );
66  
67          File outputFile = new File( getBasedir(), "target/native/" + "fileName" );
68          assertArrayEquals( new String[] { "-o", outputFile.getAbsolutePath(), "-classpath",
69                  "path1" + File.pathSeparator + "path2", "className1", "className2" }, cl.getArguments() );
70      }
71  
72      public void testWorkingDirectory()
73          throws Exception
74      {
75          JavahExecutable javah = new JavahExecutable();
76  
77          File workingDirectory = new File( getBasedir() );
78          config.setWorkingDirectory( workingDirectory );
79  
80          Commandline cl = javah.createJavahCommand( config );
81  
82          assertEquals( workingDirectory, cl.getWorkingDirectory() );
83      }
