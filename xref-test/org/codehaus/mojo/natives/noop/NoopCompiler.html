1   package org.codehaus.mojo.natives.noop;
2   
3   import java.io.File;
4   import java.util.ArrayList;
5   import java.util.List;
6   
7   import org.codehaus.mojo.natives.NativeBuildException;
8   import org.codehaus.mojo.natives.compiler.Compiler;
9   import org.codehaus.mojo.natives.compiler.CompilerConfiguration;
10  import org.codehaus.plexus.util.FileUtils;
11  
12  /**
13   * Helper class to test native-maven-plugin
14   */
15  public class NoopCompiler
16      implements Compiler
17  {
18  
19      public List compile( CompilerConfiguration config, File[] sourceFiles )
20          throws NativeBuildException
21      {
22          List compilerOutputFiles = new ArrayList( sourceFiles.length );
23  
24          for ( int i = 0; i < sourceFiles.length; ++i )
25          {
26              File source = new File( sourceFiles[i].toString() );
27  
28              File objectFile = this.getObjectFile( source, config );
29  
30              compilerOutputFiles.add( objectFile );
31  
32          }
33  
34          return compilerOutputFiles;
35      }
36  
37      /**
38       * @return
39       */
40      protected String getObjectFileExtension()
41      {
42          // no need to test system specific extension
43          return "o";
44      }
45  
46      /**
47       * Figure out the object file path from a given source file
48       *
49       * @param sourceFile
50       * @return
51       */
52      private File getObjectFile( File sourceFile, CompilerConfiguration config )
53      {
54          String srcPath = sourceFile.getPath();
55  
56          String destPath =
57              config.getOutputDirectory().getPath() + "/" + FileUtils.basename( srcPath ) + this.getObjectFileExtension();
58  
59          return new File( destPath );
60      }
61  
62  }
