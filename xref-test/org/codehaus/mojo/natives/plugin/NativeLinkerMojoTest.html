17  public class NativeLinkerMojoTest
18      extends AbstractMojoTestCase
19  {
20      private NativeLinkMojo getMojo() throws Exception {
21          File pluginXml = new File( getBasedir(), "src/test/resources/linker/plugin-config.xml" );
22          NativeLinkMojo mojo = (NativeLinkMojo) lookupMojo( "link", pluginXml );
23          assertNotNull( mojo );
24  
25          // must init this
26          mojo.setPluginContext( new HashMap() );
27          return mojo;
28      }
29  
30      public void testExecute() throws Exception
31      {
32          NativeLinkMojo mojo = getMojo();
33  
34          // simulate object files
35          List objectList = new ArrayList();
36          objectList.add( new File( "o1.o" ) );
37          objectList.add( new File( "o2.o" ) );
38          mojo.saveCompilerOutputFilePaths( objectList );
39  
40          // simulate artifact
41          ArtifactHandler artifactHandler = new DefaultArtifactHandler();
42  
43          Artifact artifact = new DefaultArtifact( "test", "test", VersionRange.createFromVersion( "1.0-SNAPSHOT" ), "compile", "exe", null, artifactHandler);
44          mojo.getProject().setArtifact( artifact );
45  
46          // simulate artifacts
47          mojo.getProject().setArtifacts( new HashSet() ); // no extern libs for now
48  
49          String linkerFinalName = "some-final-name";
50          setVariableValueToObject( mojo, "linkerFinalName", linkerFinalName );
51  
52          mojo.execute();
53  
54          LinkerConfiguration conf = mojo.getLgetLinkerConfiguration();
55  
56          // "target is set in the stub
57          assertEquals( new File( "target" ), conf.getOutputDirectory() );
58          assertEquals( linkerFinalName, conf.getOutputFileName() );
59          assertNull(conf.getOutputFileExtension() );
60          // current artifactHandler mocking return null extension name
61          assertEquals( new File( "target/some-final-name.null" ), conf.getOutputFile() );
62  
63      }
