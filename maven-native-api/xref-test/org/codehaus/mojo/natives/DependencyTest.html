43      /**
44       * Source has includes, but include path is not given
45       **/
46      public void testNoneParticipateDepedencyAnalysisInclude()
47          throws IOException, InterruptedException
48      {
49          String testSrcDir = "target/test/testNoneParticipateDepedencyAnalysisInclude/c/";
50          String testSource = testSrcDir + "test1.c";
51  
52          this.rmDir( testSrcDir );
53          this.mkDir( testSrcDir );
54  
55          this.writeFile( testSource, "#include \"test1.h\"\n \"test2.h\"\n" );
56  
57          File[] includePaths = new File[0];
58  
59          File srcFile = new File( testSource );
60  
61          Parser parser = new CParser();
62  
63          Dependency dependency = new Dependency( srcFile, parser, includePaths );
64  
65          dependency.analyze();
66  
67          assertEquals( 0, dependency.getDependencies().size() );
68  
69      }

96      /*
97       * Makesure depedencies tree does not contain duplicate node
98       */
99      public void testCyclicDependencyAnalysis()
100         throws IOException, InterruptedException
101     {
102         String testSrcDir = "target/test/testCyclicDependencyAnalysis/c/";
103         String testIncDir = "target/test/testCyclicDependencyAnalysis/h/";
104 
105         this.rmDir( testSrcDir );
106         this.rmDir( testIncDir );
107 
108         this.mkDir( testSrcDir );
109         this.mkDir( testIncDir );
110 
111         this.writeFile( testSrcDir + "test1.h", "#include \"test2.h\"" );
112 
113         Thread.sleep( 100 );
114 
115         // force a cyclic condiction
116         this.writeFile( testIncDir + "test2.h", "#include \"test1.h\"" );
117 
118         this.writeFile( testIncDir + "test3.h", "#include \"test1.h\"" );
119 
120         Thread.sleep( 100 );
121 
122         this.writeFile( testSrcDir + "test1.c", "#include \"test1.h\"\n#include \"test3.h\"" );
123 
124         File srcFile = new File( testSrcDir + "test1.c" );
125 
126         File[] includePaths = new File[2];
127 
128         includePaths[0] = new File( testSrcDir );
129         includePaths[1] = new File( testIncDir );
130 
131         Parser parser = new CParser();
132 
133         Dependency dependency = new Dependency( srcFile, parser, includePaths );
134 
135         dependency.analyze();
136 
137         assertEquals( 2, dependency.getDependencies().size() );
138 
139         assertEquals( 3, dependency.getDeepDependencyCount() );
140 
141         assertTrue( srcFile.lastModified() == dependency.getCompositeLastModified() );
142 
143     }
